$ ghci
> :load Slice.hs 
> let list = [0,1,2,3,4,5]
> sliceAround 1 list
[0,1,2]
> sliceAround 4 list
[3,4,5]
> sliceAround 0 list
[0,1]
> sliceAround 5 list
[4,5]
